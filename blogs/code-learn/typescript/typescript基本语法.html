    # 4 .成员修饰符

    • public 允许在类的内外被调用，如果不写的话 ,默认是public
    • private 允许在类内被调用
    • protected 允许在类内及继承的子类中使用
    1. 只读属性

    只读属性可以在构造器中修改，赋值之后不可以修改

    属性本身不能修改，但如果readonly修饰该对象时，里面的属性可以修改

    class People {
    readonly name: string='默认'
    readonly friend = {
        name: ''
    }
    constructor(name: string, friend: { name: string } = { name: 'hhh' }) {
        // 可以在构造器中修改属性
        this.name = name
        this.friend = friend
    }
}

const p = new People('lina')
console.log(p.name);
// 如果readonly修饰该对象时，里面的属性可以修改
p.friend.name = 'fff'
console.log(p.friend.name);

    # 5. 封装性— getters/setters

    class Person {
    private _name: string
    private _age: number
    private _sex: string

    constructor(name: string, age: number, sex: string) {
        this._name = name
        this._age = age
        this._sex = sex
    }
    getValue() {
        console.log(this._name, this._age, this._sex);
    }
    setValue(name: string, age?: number, sex?: string) {
        this._name = name
        // 可选链不能在属性赋值时使用 //name?.length
        if (age) {
            this._age = age
        }
        if (sex) {
            this._sex = sex
        }
    }
}
const p = new Person('lili', 18, 'female')
p.setValue('lisa', 20)
p.getValue()

    # 7. 抽象类

    抽象方法必须在抽象类中

    一个类继承抽象类要重写里面的抽象方法

    // 抽象类
abstract class Shape{
    abstract getArea(): number
    getName() {
        console.log('hhhh');
    }
}

class Rectangle extends Shape{
    private height: number
    private width: number
    constructor(height: number, width: number) {
        super()
        this.height = height
        this.width = width
    }
    getArea() {
        return this.height*this.width
    }  
}
class Circle extends Shape{
    private r: number
    constructor(r: number) {
        super()
        this.r = r
    }
    getArea(): number {
        return 3.14*this.r*this.r
    }
}
const re = new Rectangle(3, 4)
const area1 = re.getArea()
console.log(area1);
const cir = new Circle(3)
const area2 = cir.getArea()
console.log(area2);
